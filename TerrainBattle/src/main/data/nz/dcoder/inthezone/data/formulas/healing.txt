// Parameters relating to the agent that performs the attack:
//   c                       // spell strength
//   agent.strength
//   agent.intelligence
//   agent.level
//   agent.physicalWeapon
//   agent.magicalWeapon
//
// Parameters relating to the target:
//   target.guard
//   target.spirit
//   target.level
//   target.physicalArmour   // total of all physical armour
//   target.magicalArmour    // total of all magical armour
//
// Extras:
//    rnd                    // a random number in the range [0..1]
//

a = 1
b = 3
d = 5
g = 4
h = 2
j = 1

(0.9 + (0.2 * rnd))
  * c * (
      ((d * agent.level) / b)
    + ((h * agent.magicalWeapon) / j)
    + ((g * agent.intelligence) / a))
